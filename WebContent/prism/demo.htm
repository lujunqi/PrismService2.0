  • 遍历数组

    遍历数组回填入元素中

    javascript代码 
    var obj = [{"name":"张三","sex":"M"},{"name":"李四","sex":"M"},{"name":"王妞","sex":"F"}];
var cmd = new prismTemplete();
cmd.data("d",obj);
cmd.sex = function(obj,val){
obj.html("xxx"+val)
}
cmd.preview($("#demo2"));
    html代码 
    <div id="demo2" data-exp="d" data-method="list" data-map="map">
    <span>[<a data-exp="map.name"></a>]</span>
    </div>

  • 设置元素属性

    javascript代码 
    var obj = {};

obj["val"] = "自定义value";
var cmd = new prismTemplete();
cmd.data("d",obj);
cmd.preview($("#demo3"));
    html代码 
    <input id="demo3" data-exp="d" data-method="attr"

  • 自定义函数

    javascript代码 
    var obj = {};
obj["name"] = "张三";
obj["sex"] = "M";
var cmd = new prismTemplete();
cmd.getSex = function(obj,val){
	if(val=="F"){
		obj.val("女");
	}else{
		obj.val("男");
	}
};
cmd.data("d",obj);
cmd.preview($("#demo4"));
    html代码 
    <input id="demo4" data-exp="d" data-method="getSex">
